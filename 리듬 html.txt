function memberChecks(memberId){ return CACHE.checks.filter(c=>c.member_id===memberId); }

function weekRange(){
  const d=new Date(); const day=(d.getDay()+6)%7;
  const mon=new Date(d); mon.setDate(d.getDate()-day); mon.setHours(0,0,0,0);
  const sun=new Date(mon); sun.setDate(mon.getDate()+6);
  const f=x=>x.getFullYear()+'-'+pad2(x.getMonth()+1)+'-'+pad2(x.getDate());
  return { start:f(mon), end:f(sun) };
}
function weeklyDone(memberId){
  const {start,end}=weekRange();
  return memberChecks(memberId).filter(c=>c.check_type!=='morning' && c.response==='completed' && c.check_date>=start && c.check_date<=end).length;
}
function weeklyGoal(m){ return Math.max(1,(m.preferred_days||[]).length); }

function isTodayTarget(m){
  if (m.reminder_level==='light') return nowDow()===0;
  return (m.preferred_days||[]).includes(nowDow());
}
function todayCheckFor(memberId, type){
  const tk=todayKey();
  return memberChecks(memberId).find(c=>c.check_date===tk && c.check_type===type) || null;
}
function last3(memberId){

function recentNoResponseStreak(memberId){
  const list = memberChecks(memberId).filter(c=>c.check_type!=='morning').slice(0,3);

}
function careReason(m){
  if (m.track_enabled===false) return null;
  const t = todayCheckFor(m.id, m.reminder_level==='light'?'weekly':'evening');
  if (t && t.response==='difficult') return { key:'difficult', label:'오늘은 어려웠어요' };
  if (t && !t.response && t.sent_at) return { key:'no_response', label:'미응답' };
  if (recentNoResponseStreak(m.id) >= 2) return { key:'repeated_no_response', label:'반복 미응답' };
  return null;
}

async function doSignup(){

  const pw=document.getElementById('su-pw').value;

async function generateTodayChecks(){

async function obFinish(){

  });

}

/* ── 이번 주 요약 ── */
function viewSummary(){
  const {start,end} = weekRange();

  const difficult = evChecks.filter(c=>c.response==='difficult').length;
  const noresp = evChecks.filter(c=>c.sent_at && !c.response).length;

        <p class="mut" style="line-height:1.6">${good?'기록해둘게요.':'무리하지 않는 선에서\n흐름이 끊기지 않게 도와드릴게요.'}</p>
      </div></div>`;
  }
  const q = kind==='weekly'
    ? `이번 주 운동 흐름은<br>어떠셨나요?`
    : `${esc(m?.name||'')}님,<br>오늘 목표하신 운동,<br>잘 다녀오셨나요?`;
  const sub = kind==='weekly'
    ? '완벽하지 않아도 괜찮아요.\n이번 주에 조금이라도 움직이셨다면 기록해두고,\n어려우셨다면 다음 주에 다시 이어가면 됩니다.'
    : '무리하지 않아도 괜찮아요.\n다녀오셨다면 기록해두고,\n어려우셨다면 다음 흐름으로 이어가면 됩니다.';
  const b1 = kind==='weekly' ? '움직였어요' : '다녀왔어요';
  const b2 = kind==='weekly' ? '이번 주는 어려웠어요' : '오늘은 어려웠어요';
  return `<div class="page" style="padding-top:56px;text-align:center">
    <h1 style="margin-bottom:10px;line-height:1.4">${q}</h1>
    <p class="mut" style="margin-bottom:26px;white-space:pre-line;line-height:1.6">${sub}</p>
    <button class="ansbtn a1" onclick="submitCheck('completed')">${b1}</button>
    <button class="ansbtn a2" onclick="submitCheck('difficult')">${b2}</button>
  </div>`;
}
